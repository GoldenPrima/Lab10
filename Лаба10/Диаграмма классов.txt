title Class Diagram: University Information System

class University {
  - faculties: List<Faculty>
  + add_faculty(faculty: Faculty)
}

class Faculty {
  - name: String
  - dean: Dean
  - institutes: List<Institute>
  + set_dean(dean: Dean)
  + add_institute(institute: Institute)
}

class Institute {
  - name: String
  - address: String
}

abstract class Employee {
  - ss_no: int
  - name: String
  - email: String
  # static counter: int
  + {static} getCounter(): int
}

class AdministrativeEmployee {
}

class ResearchEmployee {
  - field_of_study: String
  - institute: Institute
  - projects: List<Participation>
  - courses: List<Course>
  + add_project(participation: Participation)
  + add_course(course: Course)
}

class Lecturer {
}

class Dean {
}

class Project {
  - name: String
  - start_date: Date
  - end_date: Date
}

class Participation {
  - hours: int
  - employee: ResearchEmployee
  - project: Project
}

class Course {
  - course_id: int
  - name: String
  - weekly_duration: float
}
